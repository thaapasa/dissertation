\begin{listofsymbols}

\loxline{\comver}{The latest committed version of a multiversion
database index}{def:maxcomver}

\loxline{\actver}{The active version of a TMVBT index. 
Updates to the index receive this version}{def:actver}

\loxline{\stablever}{The most recent version that has been moved to the
TMVBT from the VBT in a CMVBT index}{def:stablever}

\loxline{\movever}{The version currently being moved from the
VBT into the TMVBT in a CMVBT index}{def:movever}

\loxline{\txid{T}}{The unique identifier of an active
transaction~$T$}{def:txid}

\loxline{\snapver{T}}{The snapshot version of transaction~$T$; i.e., the
version that $T$ is reading}{def:snapver}

\loxline{\commitver{T}}{The unique commit-time version assigned to
transaction~$T$ upon committing}{def:commitver}

\loxline{\capacity}{The logical capacity of a database page, i.e., how many
entries fit to a page}{def:capacity}

\loxline{\entries{v}}{The number of entries in the database that are alive
at version~$v$}{def:entries}

\loxline{\deletemark}{A special value used to record key

\loxline{\rootstar}{A search structure that stores the page identifiers of
the root pages of a multi-root multiversion index, indexed by the root page 
versions}{def:rootstar}
